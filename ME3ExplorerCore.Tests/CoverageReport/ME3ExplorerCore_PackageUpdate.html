 487        /// <param name="uncompressedStream"></param>

 490        {

 504            //uint CompressedSize

 530            {

 532                {

 546                }
 547            }
 548            //Debug.WriteLine($"Chunk {chunkNum} contains {firstElement} to {lastElement}");
 549            chunks.Add(chunk);

 551            //Rewrite header with chunk table information so we can position the data blocks after table
 552            compressedStream.Position = 0;
 553            package.WriteHeader(compressedStream, compressionType, chunks);
 554            MemoryStream m1 = new MemoryStream();

 556            for (int c = 0; c < chunks.Count; c++)
 557            {

 560                chunk.compressedSize = 0; // filled later

 563                int numBlocksInChunk = (int)Math.Ceiling(chunk.uncompressedSize * 1.0 / CompressionHelper.MAX_BLOCK_SIZE
 564                // skip chunk header and blocks table - filled later
 565                compressedStream.Seek(CompressionHelper.SIZE_OF_CHUNK_HEADER + CompressionHelper.SIZE_OF_CHUNK_BLOCK_HEA
 566
 567                uncompressedStream.JumpTo(chunk.uncompressedOffset);
 568
 569                chunk.blocks = new List<CompressionHelper.Block>();
 570
 571                //Calculate blocks by splitting data into 128KB "block chunks".
 572                for (int b = 0; b < numBlocksInChunk; b++)
 573                {
 574                    CompressionHelper.Block block = new CompressionHelper.Block();
 575                    block.uncompressedsize = Math.Min(CompressionHelper.MAX_BLOCK_SIZE, dataSizeRemainingToCompress);
 576                    dataSizeRemainingToCompress -= block.uncompressedsize;
 577                    block.uncompressedData = uncompressedStream.ReadToBuffer(block.uncompressedsize);
 578                    chunk.blocks.Add(block);
 579                }
 580
 581                if (chunk.blocks.Count != numBlocksInChunk) throw new Exception("Number of blocks does not match expecte
 582
 583                //Compress blocks
 584                Parallel.For(0, chunk.blocks.Count, b =>
 585                {
 586                    CompressionHelper.Block block = chunk.blocks[b];
 587                    if (compressionType == CompressionType.LZO)
 588                        block.compressedData = LZO2.Compress(block.uncompressedData);
 589                    else if (compressionType == CompressionType.Zlib)
 590                        block.compressedData = Zlib.Compress(block.uncompressedData);
 591                    else
 592                        throw new Exception("Internal error: Unsupported compression type for compressing blocks: " + co
 593                    if (block.compressedData.Length == 0)
 594                        throw new Exception("Internal error: Block compression failed! Compressor returned no bytes");
 595                    block.compressedsize = (int)block.compressedData.Length;
 596                    chunk.blocks[b] = block;
 597                });
 598
 599                //Write compressed data to stream
 600                for (int b = 0; b < numBlocksInChunk; b++)
 601                {
 602                    var block = chunk.blocks[b];
 603                    compressedStream.Write(block.compressedData, 0, (int)block.compressedsize);
 604                    chunk.compressedSize += block.compressedsize;
 605                }
 606                chunks[c] = chunk;
 607            }
 608
 609            //Update each chunk header with new information
 610            foreach (var c in chunks)
 611            {
 612                compressedStream.JumpTo(c.compressedOffset); // jump to blocks header
 613                compressedStream.WriteUInt32(packageTagLittleEndian);
 614                compressedStream.WriteUInt32(CompressionHelper.MAX_BLOCK_SIZE); //128 KB
 615                compressedStream.WriteInt32(c.compressedSize);
 616                compressedStream.WriteInt32(c.uncompressedSize);
 617
 618                //write block header table
 619                foreach (var block in c.blocks)
 620                {
 621                    compressedStream.WriteInt32(block.compressedsize);
 622                    compressedStream.WriteInt32(block.uncompressedsize);
 623                }
 624            }
 625
 626            //Write final header
 627            compressedStream.Position = 0;
 628            package.WriteHeader(compressedStream, compressionType, chunks);
 629            return compressedStream;
 630        }
